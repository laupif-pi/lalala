import random
def caishu():
    i = 0
    key = random.randint(1, 10)
    while i < 5:
        guss = int(input("Enter:"))
        if guss == key:
            print("good guss!")
            break
        elif guss > key:
            print("大了")
        else:
            print("小了")
        i = i + 1
    else:
        print("game over！")
        print("The key is %d" % key)
caishu()
